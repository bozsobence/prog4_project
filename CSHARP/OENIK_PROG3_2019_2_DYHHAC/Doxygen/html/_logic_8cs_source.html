540  public IEnumerable<ResultClasses.DailyIncomeResult> GetDailyIncome()
541  {
542  List<ResultClasses.DailyIncomeResult> results = new List<ResultClasses.DailyIncomeResult>();
543  for (int i = 1; i <= 12; i++)
544  {
545  var daily = from x in this.rentRepo.GetAll()
546  where x.StartTime.Month == i
547  group x by x.StartTime.Day into g
548  select new ResultClasses.DailyIncomeResult
549  {
550  Month = i,
551  Day = g.Key,
552  Income = (int)g.Sum(x => x.Price),
553  };
554  if (daily.Count() != 0)
555  {
556  foreach (var day in daily)
557  {
558  results.Add(day);
559  }
560  }
561  }
562 
563  return results;
564  }

582  public ResultClasses.RentsByUserResult GetUserWithMostRents()
583  {
