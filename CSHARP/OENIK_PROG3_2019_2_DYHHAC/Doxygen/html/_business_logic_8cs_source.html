541  public IEnumerable<DailyIncomeResult> GetDailyIncome()
542  {
543  List<DailyIncomeResult> results = new List<DailyIncomeResult>();
544  for (int i = 1; i <= 12; i++)
545  {
546  var daily = from x in this.rentRepo.GetAll()
547  where x.StartTime.Month == i
548  group x by x.StartTime.Day into g
549  select new DailyIncomeResult
550  {
551  Month = i,
552  Day = g.Key,
553  Income = (int)g.Sum(x => x.Price),
554  };
555  if (daily.Count() != 0)
556  {
557  foreach (var day in daily)
558  {
559  results.Add(day);
560  }
561  }
562  }
563 
564  return results;

583  public RentsByUserResult GetUserWithMostRents()
584  {
585  var rentData = this.rentRepo.GetAll();
586  var accData = this.accountRepo.GetAll();

588  join account in accData
589  on rent.AccountId equals account.AccountId
590  group rent by rent.AccountId into g
591  select new

593  ID = g.Key,
594  RENTS = g.Count(),
595  }).OrderByDescending(x => x.RENTS).FirstOrDefault();
596  var mostRents = accData.Where(x => x.AccountId == rents.ID).FirstOrDefault().Name;

683  {
684  carCategory = "norml";
